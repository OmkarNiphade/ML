import pandas as pd
from sklearn.linear_model import LinearRegression
data = pd.DataFrame({
 'area':[1000,1500,2000],
 'bedroom':[2,3,4],
 'price':[10,15,20]
})
X = data[['area','bedroom']]
y = data['price']
model = LinearRegression()
model.fit(X, y)
print("Predicted price =", model.predict([[1800,3]])[0])
