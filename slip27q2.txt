import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import PolynomialFeatures

data = pd.read_csv("Salary_positions.csv")

X = data[["Level"]]
y = data["Salary"]

lin = LinearRegression()
lin.fit(X, y)

poly = PolynomialFeatures(degree=4)
X_poly = poly.fit_transform(X)
poly_model = LinearRegression()
poly_model.fit(X_poly, y)

print("Simple Linear Regression Score:", lin.score(X, y))
print("Polynomial Regression Score:", poly_model.score(X_poly, y))

print("Level 11 Linear:", lin.predict([[11]])[0])
print("Level 12 Linear:", lin.predict([[12]])[0])

print("Level 11 Polynomial:", poly_model.predict(poly.transform([[11]]))[0])
print("Level 12 Polynomial:", poly_model.predict(poly.transform([[12]]))[0])
